import os
import subprocess

def git_push_script():
    folder_path = input("Enter the folder path where your Git project is located: ")
    
    if os.path.isdir(folder_path):
        os.chdir(folder_path)
    else:
        print(f"The path {folder_path} is invalid.")
        return
    
    origin_link = input("Enter the GitHub origin URL: ")

    subprocess.run(["git", "remote", "rm", "origin"], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    subprocess.run(["git", "remote", "add", "origin", origin_link], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    
    subprocess.run(["git", "add", "."], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    
    commit_message = input("Enter commit message: ")
    subprocess.run(["git", "commit", "-m", commit_message], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    
    subprocess.run(["git", "push", "origin", "master"], stdout=subprocess.PIPE, stderr=subprocess.PIPE)

    print("Push complete!")

def ui():
    print("Welcome to the Git Management Tool")
    print("1. Git Push")
    print("2. Check Git Status")
    choice = input("Please choose an option (1 or 2): ")

    if choice == "1":
        git_push_script()
    elif choice == "2":
        folder_path = input("Enter the folder path where your Git project is located: ")
        
        if os.path.isdir(folder_path):
            os.chdir(folder_path)
            status = subprocess.run(["git", "status"], capture_output=True, text=True)
            print(status.stdout)
        else:
            print(f"The path {folder_path} is invalid.")
    else:
        print("Invalid choice. Please enter 1 or 2.")

if __name__ == "__main__":
    ui()
